           )
         AND (rapidsms_httprouter_message.connection_id = 
                 (
                  SELECT rapidsms_connection.id

                 LIMIT 1
                 )
           )
     )

,
    
(
SELECT DISTINCT count
     (poll_poll_contacts.id) AS count
           
  FROM poll_poll_contacts
          
 WHERE (poll_poll_contacts.contact_id = contact.id)
          
GROUP BY poll_poll_contacts.contact_id
) AS questions
,
    
(
SELECT count
     (poll_response.id) AS count
           
  FROM poll_response
          
 WHERE (poll_response.contact_id = contact.id)

(
SELECT message.text

        JOIN poll_response response

                 (response.message_id = message.id)
           )
     )
     
 WHERE (
           (
                 (response.poll_id = 121)
               AND (response.has_errors = false)
           )
         AND (response.contact_id = contact.id)
     )
    LIMIT 1
) AS source
,
    contact.is_caregiver

           )
           ,', '::text
